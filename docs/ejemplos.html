EJEMPLO 1 FILTRO FIR

FIR Pasa-Bajas para ECG

Elimina el ruido muscular (EMG) y la interferencia de alta frecuencia de una señal ECG. Frecuencia de corte 40 Hz a 960 Hz de muestreo. Diseñado con ventana Hamming, 51 coeficientes.

Resultado esperado en Serial Plotter:
  • Canal 1 (raw): señal ruidosa con picos agudos
  • Canal 2 (filtrada): forma de onda suave, retardo de 25 muestras visible
terminalSerial_results_BioFilterLib_FIR.ino
/*
 * BioFilterLib - Ejemplo FIR Pasa-Bajas ECG
 * Señal: ECG real cargada desde Waveforms.h (Fs = 960 Hz)
 * Filtro: Pasa-bajas 40 Hz, 51 taps, Ventana Hamming
 * Salida: Serial Plotter (raw,filtrada)
 */
#include <BioFilterLib.h>
#include "Waveforms.h"   // Señales de prueba incluidas en src/utils/

// Coeficientes FIR pasa-bajas 40 Hz @ 960 Hz, 51 taps, Hamming
const float32_t firCoeffs[51] = {
  -7.22e-4f, -1.27e-3f, -1.54e-3f, -9.96e-4f,  7.38e-4f,
   3.84e-3f,  8.02e-3f,  1.25e-2f,  1.57e-2f,  1.58e-2f,
   1.06e-2f,  0.00e+0f, -1.51e-2f, -3.18e-2f, -4.65e-2f,
  -5.47e-2f, -5.20e-2f, -3.58e-2f,  0.00e+0f,  4.98e-2f,
   1.08e-1f,  1.68e-1f,  2.22e-1f,  2.61e-1f,  2.80e-1f,
   2.61e-1f,  2.22e-1f,  1.68e-1f,  1.08e-1f,  4.98e-2f,
   0.00e+0f, -3.58e-2f, -5.20e-2f, -5.47e-2f, -4.65e-2f,
  -3.18e-2f, -1.51e-2f,  0.00e+0f,  1.06e-2f,  1.58e-2f,
   1.57e-2f,  1.25e-2f,  8.02e-3f,  3.84e-3f,  7.38e-4f,
  -9.96e-4f, -1.54e-3f, -1.27e-3f, -7.22e-4f,  0.00e+0f,
   0.00e+0f
};

const uint16_t NUM_TAPS   = 51;
const uint16_t BLOCK_SIZE = 1;
const uint16_t GROUP_DELAY = (NUM_TAPS - 1) / 2; // 25 muestras

FIRFilter filtroFIR(firCoeffs, NUM_TAPS, BLOCK_SIZE);

// Buffer circular para compensar el retardo de grupo
float32_t delayBuf[51] = {0};
uint16_t delayIdx = 0;

void setup() { Serial.begin(115200); }

void loop() {
  static uint16_t idx = 0;
  if (idx >= NUM_SAMPLES) idx = 0;

  // Cargar señal ECG desde flash (Waveforms.h)
  float32_t raw = loadSignal(SIGNAL_ECG_NOISY, idx++);

  // Filtrar
  float32_t filtrada = filtroFIR.processSample(raw);

  // Alinear raw con la señal filtrada (compensar retardo)
  delayBuf[delayIdx] = raw;
  float32_t rawAlineado = delayBuf[(delayIdx - GROUP_DELAY + 51) % 51];
  delayIdx = (delayIdx + 1) % 51;

  Serial.print(rawAlineado, 5); Serial.print(",");
  Serial.println(filtrada,  5);
  delay(1); // ~960 Hz con Serial overhead
}

EJEMPLO 2 FILTRO IIR

IIR Notch 60 Hz para ECG

Elimina la interferencia de la red eléctrica (60 Hz) con una sola sección biquad. Factor de calidad Q=30 produce una muesca muy estrecha sin afectar la morfología del ECG.

Por qué Q=30:

Un Q alto (ancho de banda estrecho = 60/30 = 2 Hz) elimina solo 59–61 Hz, sin afectar componentes cardíacas importantes. Un Q bajo eliminaría demasiado del espectro.

terminalSerial_results_BioFilterLib_IIR.ino
/*
 * BioFilterLib - Ejemplo IIR Notch 60 Hz
 * Filtro biquad de 2do orden, Q=30, Fs=960 Hz
 * Coeficientes generados con: scipy.signal.iirnotch(60, 30, 960)
 * IMPORTANTE: CMSIS invierte el signo de a1 y a2 vs scipy
 */
#include <BioFilterLib.h>
#include "Waveforms.h"

// Coeficientes Notch 60 Hz, Q=30 — formato CMSIS {b0,b1,b2,a1,a2}
// scipy: b=[0.99034,-1.80193,0.99034], a=[1,-1.80193,0.98068]
// CMSIS: a1 = -(-1.80193) = +1.80193,  a2 = -(0.98068) = -0.98068
const float32_t notchCoeffs[5] = {
   0.99034f,   // b0
  -1.80193f,   // b1
   0.99034f,   // b2
   1.80193f,   // a1 (negado)
  -0.98068f    // a2 (negado)
};

IIRFilter notch60Hz(notchCoeffs, 1, 1);

void setup() { Serial.begin(115200); }

void loop() {
  static uint16_t idx = 0;
  if (idx >= NUM_SAMPLES) idx = 0;

  // ECG con ruido de 60 Hz superpuesto
  float32_t raw      = loadSignal(SIGNAL_ECG_NOISY_60HZ, idx++);
  float32_t filtrada = notch60Hz.processSample(raw);

  // IIR tiene retardo mínimo: no se necesita compensación
  Serial.print(raw,     5); Serial.print(",");
  Serial.println(filtrada, 5);
  delay(1);
}

EJEMPLO 3 FILTRO LMS ADAPTATIVO

LMS — Cancelación Adaptativa de Ruido

Este ejemplo genera un ECG sintético con onda P, complejo QRS y onda T, le suma interferencia de 60 Hz, y usa el LMS para cancelarla adaptativamente. Permite visualizar la convergencia del algoritmo en tiempo real.

4 señales en Serial Plotter:
  1. ecgContaminado — ECG + 60 Hz (señal de entrada)
  2. ref60Hz — seno de referencia puro
  3. ecgLimpio — salida del LMS (converge progresivamente)
  4. errorAdaptacion — decae a cero al converger
terminalSerial_results_BioFilterLib_LMS.ino
/*
 * BioFilterLib - Ejemplo LMS Adaptativo
 * ECG sintético (75 BPM) + interferencia 60 Hz
 * LMS: 64 taps, mu=0.01
 * Observar en Serial Plotter cómo el error decrece con el tiempo
 */
#include <BioFilterLib.h>

const float FS     = 960.0f;
const float BPM    = 75.0f;
const float F_NOISE = 60.0f;
const float A_NOISE = 0.3f;  // Amplitud del ruido 60 Hz

// LMS: 64 taps, mu=0.01, blockSize=1
const uint16_t LMS_TAPS = 64;
float32_t lmsCoeffs[LMS_TAPS] = {0.0f};
LMSFilter cancelador(lmsCoeffs, LMS_TAPS, 0.01f, 1);

float t = 0.0f;

// Genera un latido ECG sintético simple
float32_t ecgSintetico(float tiempo) {
  float phase = fmod(tiempo * BPM / 60.0f, 1.0f);
  float ecg = 0.0f;
  // Onda P
  ecg += 0.15f * exp(-0.5f * pow((phase - 0.2f) / 0.04f, 2));
  // Onda Q
  ecg -= 0.1f  * exp(-0.5f * pow((phase - 0.36f) / 0.008f, 2));
  // Onda R (pico principal)
  ecg += 1.0f  * exp(-0.5f * pow((phase - 0.4f)  / 0.006f, 2));
  // Onda S
  ecg -= 0.2f  * exp(-0.5f * pow((phase - 0.44f) / 0.008f, 2));
  // Onda T
  ecg += 0.35f * exp(-0.5f * pow((phase - 0.6f)  / 0.04f,  2));
  return ecg;
}

void setup() { Serial.begin(115200); }

void loop() {
  // ECG sintético + ruido 60 Hz
  float32_t ecg           = ecgSintetico(t);
  float32_t ruido60Hz     = A_NOISE * sin(2.0f * PI * F_NOISE * t);
  float32_t ecgContaminado = ecg + ruido60Hz;

  // Referencia: seno puro de 60 Hz (sin el ECG)
  float32_t ref60Hz = sin(2.0f * PI * F_NOISE * t);

  // Filtrado adaptativo
  float32_t ecgLimpio, errorAdaptacion;
  cancelador.processSample(ecgContaminado, ref60Hz,
                           &ecgLimpio, &errorAdaptacion);

  Serial.print(ecgContaminado,  4); Serial.print(",");
  Serial.print(ref60Hz,         4); Serial.print(",");
  Serial.print(ecgLimpio,       4); Serial.print(",");
  Serial.println(errorAdaptacion, 4);

  t += 1.0f / FS;
  delayMicroseconds(1042);
}

EJEMPLO 4 WAVELET DB-4

Wavelet DB-4 — Descomposición y Reconstrucción

Descompone la señal en aproximación (baja frecuencia) y detalle (alta frecuencia), y reconstruye versiones filtradas. La carpeta de ejemplos incluye capturas de pantalla del Serial Plotter para DB-4 y DB-8.

terminalSerial_results_BioFilterLib_Wavelet.ino
/*
 * BioFilterLib - Ejemplo Wavelet DB-4
 * Descomposición en aproximación (LP) y detalle (HP)
 * Reconstrucción con solo aproximación = denoising
 * Salida: 4 señales en Serial Plotter
 */
#include <BioFilterLib.h>
#include "Waveforms.h"

const uint32_t BUF_SIZE = 256;

// Buffers para procesamiento por bloques
float32_t inputBuf[BUF_SIZE];
float32_t approxBuf[BUF_SIZE];
float32_t detailBuf[BUF_SIZE];

WaveletFilter wavelet(BUF_SIZE); // blockSize = 256 (por lotes)

void setup() {
  Serial.begin(115200);

  // Cargar un bloque de ECG ruidoso desde flash
  for (uint32_t i = 0; i < BUF_SIZE; i++) {
    inputBuf[i] = loadSignal(SIGNAL_ECG_NOISY, i);
  }

  // Descomposición wavelet del bloque completo
  wavelet.processBuffer(inputBuf, approxBuf, detailBuf, BUF_SIZE);
}

void loop() {
  static uint32_t idx = 0;
  if (idx >= BUF_SIZE) { idx = 0; wavelet.reset(); return; }

  // Reconstrucción con solo aproximación (denoising)
  float32_t soloApprox   = wavelet.reconstruct(approxBuf[idx], 0.0f);
  // Reconstrucción completa (debe ser ≈ original)
  float32_t reconstruida = wavelet.reconstruct(approxBuf[idx], detailBuf[idx]);

  // 4 señales: original, aproximación, detalle, denoised
  Serial.print(inputBuf[idx],  4); Serial.print(",");
  Serial.print(approxBuf[idx], 4); Serial.print(",");
  Serial.print(detailBuf[idx], 4); Serial.print(",");
  Serial.println(soloApprox,   4);

  idx++;
  delay(1);
}
